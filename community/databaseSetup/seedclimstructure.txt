CREATE TABLE `sites`
 (
	`siteID`			varchar (40), 
	`norwegianName`			varchar (40), 
	`siteCode`			varchar (12), 
	`latitude`			float, 
	`longitude`			float, 
	`x_UTM33_North`			float, 
	`y_UTM33_north`			int, 
	`altitude(DEM)`			int, 
	`annualprecipitation_gridded`			float, 
	`temperature_level`			int, 
	`summerTemperature_gridded`			float, 
	`precipitation_level`			int, 
	`geology`			varchar (100), 
	`landUse`			varchar (100), 
	`aspect`			float, 
	`solar_radiation(DEM)(mw/hour/m^2)`			float, 
	`totalN (mg N/m2 yr)`			int, 
	`slope`			float,
	 PRIMARY KEY (siteID)
);


CREATE TABLE `blocks`
 (
	`blockID`			varchar (20), 
	`siteID`			varchar (40), 
	`slope`			float, 
	`aspect`			float, 
	`comment`			text (255),
	 PRIMARY KEY (blockID),
	 FOREIGN KEY (siteID) REFERENCES sites(siteID)
);

CREATE TABLE `plots`
 (
	`plotID`			varchar (10), 
	`blockID`			varchar (40), 
	`aspect`			float, 
	`slope`			float,
	 PRIMARY KEY (plotID),
	 FOREIGN KEY (blockID) REFERENCES blocks(blockID)
);

CREATE TABLE `turfs`
 (
	`turfID`			varchar (40), 
	`TTtreat`			varchar (10), 
	`originPlotID`			varchar (10), 
	`destinationPlotID`			varchar (10),
	 PRIMARY KEY (turfID),
	 FOREIGN KEY (originPlotID) REFERENCES plots(plotID)
);

CREATE TABLE `taxon`
 (
	`species`			varchar (10), 
	`speciesName`			varchar (200), 
	`authority` varchar(200),
	`family`			varchar (50), 
	`functionalGroup`			varchar (50), 
	`lifeSpan`			varchar (50), 
	`comment`			varchar (510), 
	`height`			float, 
	`leafSize`			float, 
	`seedMass`			float, 
	`SLA`			float,
	 PRIMARY KEY (species)
);


CREATE TABLE `subTurfEnvironment`
 (
	`turfID`			varchar (100), 
	`subTurf`			int, 
	`year`			int, 
	`moss`			int, 
	`lichen`			int, 
	`litter`			int, 
	`soil`			int, 
	`rock`			int, 
	`comment`			varchar (510), 
	`bad`			varchar (2),
	 PRIMARY KEY (turfID, subTurf, year),
	 FOREIGN KEY (turfID) REFERENCES turfs(turfID)
);

CREATE TABLE `turfCommunity`
 (
	`turfID`			varchar (40), 
	`year`			int, 
	`species`			varchar (100), 
	`cover`			float, 
	`cf`			int default "",
	`flag`    varchar(10) default "none",
	 PRIMARY KEY (turfID, year, species, flag),
	 FOREIGN KEY (turfID) REFERENCES turfs(turfID)

);

CREATE TABLE `turfEnvironment`
 (
	`turfID`			varchar (100), 
	`year`			int, 
	`moss`			float, 
	`lichen`			float, 
	`litter`			float, 
	`soil`			float, 
	`rock`			float, 
	`totalVascular`			float, 
	`totalBryophytes`			float, 
	`totalLichen`			float, 
	`vegetationHeight`			float, 
	`mossHeight`			float, 
	`litterThickness`			float, 
	`comment`			varchar (510), 
	`recorder`			varchar (100), 
	`date`			varchar (100),
	 PRIMARY KEY (turfID, year),
	 FOREIGN KEY (turfID) REFERENCES turfs(turfID)

);


CREATE TABLE `subTurfCommunity`
 (
	`turfID`			varchar (40), 
	`subTurf`			int, 
	`year`			int, 
	`species`			varchar (100), 
	`seedlings`			int, 
	`juvenile`			int, 
	`adult`			int NOT NULL, 
	`cf`			int,
	`flag`    varchar(10) default "none",
	 PRIMARY KEY (turfID, subTurf, year, species, flag),
	 FOREIGN KEY (turfID) REFERENCES turfs(turfID)

);
